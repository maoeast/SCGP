# Plan 04.4: CBCL UI Implementation

## Goal
Create the complete CBCL assessment UI including two-step wizard, social competence form, behavior question pagination, and clinical profile report with ECharts visualization.

## Context
CBCL has a unique two-part structure:
- Part 2: Social Competence (custom form with 7 sections)
- Part 3: 113 Behavior Questions (standard Likert, needs pagination)
- Report: Clinical profile chart with T-scores

### Task 3: Modify AssessmentContainer for Two-Step Flow
**File:** `src/views/assessment/AssessmentContainer.vue`

Add CBCL-specific two-step flow:
```typescript
// Add to AssessmentPhase type
type AssessmentPhase = 'loading' | 'welcome' | 'social' | 'assessing' | 'complete'

// Add state
const cbclStep = ref<'social' | 'behavior'>('social')
const socialFormData = ref<CBCLSocialCompetenceData | null>(null)

// Modify phase handling
const handleStartAssessment = () => {
  if (scaleCode.value === 'cbcl') {
    phase.value = 'social'
    cbclStep.value = 'social'
  } else {
    phase.value = 'assessing'
    state.value.startTime = Date.now()
  }
}

// Handle social form submission
const handleSocialFormSubmit = (data: CBCLSocialCompetenceData) => {
  socialFormData.value = data
  // Pass to driver if method exists
  if (driver.value && 'setSocialData' in driver.value) {
    (driver.value as any).setSocialData(data)
  }
  cbclStep.value = 'behavior'
  phase.value = 'assessing'
  state.value.startTime = Date.now()
}

// Add pagination for CBCL
const currentPage = ref(1)
const pageSize = 10
const totalPages = computed(() => Math.ceil(113 / pageSize))

const paginatedQuestions = computed(() => {
  if (scaleCode.value !== 'cbcl') return questions.value
  const start = (currentPage.value - 1) * pageSize
  return questions.value.slice(start, start + pageSize)
})
```

**Template additions:**
```vue
<!-- CBCL Social Form Step -->
<CBCLSocialForm
  v-else-if="phase === 'social' && scaleCode === 'cbcl'"
  :student="student"
  @submit="handleSocialFormSubmit"

<!-- Pagination for CBCL -->
<div v-if="scaleCode === 'cbcl' && phase === 'assessing'" class="pagination">
  <el-pagination
    v-model:current-page="currentPage"
    :page-size="pageSize"
    :total="113"
    layout="prev, pager, next"
    @change="handlePageChange"
  />
  <span class="page-info">第 {{ currentPage }} / {{ totalPages }} 页</span>
</div>
```

### Task 4: Implement saveCBCLAssessment Method
**File:** `src/views/assessment/AssessmentContainer.vue`

```typescript
const saveCBCLAssessment = async (
  scoreResult: ScoreResult,
  feedback: AssessmentFeedback
): Promise<number> => {
  const db = getDatabase()

  // Get CBCL-specific data from scoreResult
  const cbclData = scoreResult as CBCLScoreResult

  const result = db.run(`
    INSERT INTO cbcl_assess (
      student_id, age_months, gender,
      social_competence_data,
      social_activity_score, social_social_score, social_school_score,
      raw_answers, behavior_raw_scores, factor_t_scores,
      total_problems_score, total_problems_t_score,
      internalizing_t_score, externalizing_t_score,
      summary_level, start_time, end_time
    ) VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?)
  `, [
    student.value!.id,
    student.value!.ageInMonths,
    student.value!.gender === '男' ? 'male' : 'female',
    JSON.stringify(socialFormData.value),
    cbclData.socialCompetence?.activity.score || 0,
    cbclData.socialCompetence?.social.score || 0,
    cbclData.socialCompetence?.school.score || 0,
    JSON.stringify(scoreResult.rawAnswers),
    JSON.stringify(cbclData.behaviorProblems?.rawScores),
    JSON.stringify(cbclData.behaviorProblems?.tScores),
    scoreResult.totalScore,
    cbclData.behaviorProblems?.totalProblemsTScore || 0,
    cbclData.behaviorProblems?.internalizingTScore || 0,
    cbclData.behaviorProblems?.externalizingTScore || 0,
    scoreResult.levelCode,
    new Date(state.value.startTime).toISOString(),
    new Date().toISOString()
  ])

  return result.lastInsertRowId as number
}
```
