### Task 6: Implement Bridge Logic - Text to Count
**File:** `src/strategies/assessment/CBCLDriver.ts` or form submit handler

Add transformation method:
```typescript
function transformTextsToCounts(socialData: CBCLSocialTexts): CBCLSocialCounts {
  return {
    I_count: calculateCount(socialData.sports),
    II_count: calculateCount(socialData.hobbies),
    III_count: calculateCount(socialData.organizations),
    IV_count: calculateCount(socialData.labor),
    // ... other fields
  }
}

function calculateCount(texts: { a: string, b: string, c: string, none: boolean }): number {
  if (none) return 0
  return [texts.a, texts.b, texts.c].filter(t => t.trim().length > 0).length
}
```
